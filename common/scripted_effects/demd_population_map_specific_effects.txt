	title:$TITLE$ = {
		every_in_de_jure_hierarchy = {
			limit = { 
				tier = tier_county 
				NOT = { title_province = { has_holding_type = tribal_holding } }
				NOT = { faith = { has_doctrine = unreformed_faith_doctrine } }
			}
			every_county_province = {
				limit = { has_holding = no }
				random_list = {
					0 = {
						modifier = {
							always = yes
							add = $PERCENT$
						}
						demd_build_minor_holding = yes
					}
					100 = {
						modifier = {
							always = yes
							add = -$PERCENT$
						}
						# do nothing
					}
				}
				
			}
		}
	}
}
