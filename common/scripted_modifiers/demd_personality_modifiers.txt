# +learning/stewardship -intrigue
demd_acquire_pensive_modifier = {
	modifier = {		
		exists = real_father
		add = {
			real_father = {
				value = pensive_stat_average
				multiply = childhood_trait_parent_stat_mult
			}
		}
	}
	modifier = {		
		exists = mother		
		add = {
			mother = {
				value = pensive_stat_average
				multiply = childhood_trait_parent_stat_mult
			}
		}
	}
	modifier = {
		add = {
			value = 0
			every_relation = {
				type = guardian
				add = {
					value = pensive_stat_average
					multiply = childhood_trait_guardian_stat_mult
				}
			}
		}		
	}
	modifier = {
		culture = { has_cultural_parameter = pensive_trait_more_common }
		add = childhood_trait_culture_bonus
	}
	modifier = {
		culture = { has_cultural_parameter = pensive_trait_less_common }
		add = childhood_trait_culture_malus
	}
	modifier = {
		exists = location.province_owner
		location.province_owner = {
			has_royal_court = yes
			OR = {
				has_court_type = court_scholarly
				has_court_type = court_administrative
			}
		}
		add = childhood_trait_court_bonus
	}
	modifier = {
		exists = location.province_owner
		location.province_owner = {
			has_royal_court = yes
			has_court_type = court_intrigue
		}
		add = childhood_trait_court_malus
	}
}

demd_acquire_vengeful_modifier = {
	# Genetics
	modifier = {		
		exists = real_father	
		add = {
			value = 0
			real_father = {
				if = {
					limit = { has_trait = vengeful }
					add = parental_trait_inheritance_multiplier
				}
				else_if = {
					limit = { has_trait = forgiving }
					add = parental_opposite_trait_inheritance_multiplier
				}
			}		
		}
	}
	modifier = {		
		exists = mother		
		add = {
			value = 0
			mother = {
				if = {
					limit = { has_trait = vengeful }
					add = parental_trait_inheritance_multiplier
				}
				else_if = {
					limit = { has_trait = forgiving }
					add = parental_opposite_trait_inheritance_multiplier
				}
			}		
		}
	}
	
	# Guardians
	modifier = {
		add = {
			value = 0
			every_relation = {
				type = guardian
				limit = { has_trait = vengeful }
				add = guardian_trait_inheritance_multiplier
			}
			every_relation = {
				type = guardian
				limit = { has_trait = forgiving }
				add = guardian_opposite_trait_inheritance_multiplier
			}
		}		
	}
	
	# Faith
	modifier = {		
		add = {
			value = 0
			faith = {
				if = {
					limit = { trait_is_virtue = vengeful }
					add = faith_trait_virtue_inheritance_multiplier
				}
				else_if = {
					limit = { trait_is_sin = vengeful }
					add = faith_trait_sin_inheritance_multiplier
				}
				
				if = {
					limit = { trait_is_virtue = forgiving }
					add = faith_opposite_trait_virtue_inheritance_multiplier
				}
				else_if = {
					limit = { trait_is_sin = forgiving }
					add = faith_opposite_trait_sin_inheritance_multiplier
				}
			}
		}
	}
	
	modifier = {
	 	culture = { has_cultural_parameter = vengeful_trait_more_common }
		add = culture_trait_inheritance_multiplier
	}
	modifier = {
	 	culture = { has_cultural_parameter = vengeful_trait_less_common }
		add = culture_opposite_trait_inheritance_multiplier
	}
}
